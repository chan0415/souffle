134  /** Insert tuple */
135  virtual void insert(const RamDomain* tuple) {
136  // check for null-arity
137  if (arity == 0) {
138  // set number of tuples to one -- that's it
139  num_tuples = 1;
140  return;
141  }
142 
143  ASSERT(tuple);
144 
145  // make existence check
146  if (exists(tuple)) {
147  return;
148  }
149 
150  // prepare tail
151  if (tail->getFreeSpace() < arity || arity == 0) {
152  tail->next = std::make_unique<Block>();
153  tail = tail->next.get();
154  }
155 
156  // insert element into tail
157  RamDomain* newTuple = &tail->data[tail->used];
158  for (size_t i = 0; i < arity; ++i) {
159  newTuple[i] = tuple[i];
160  }
161  tail->used += arity;
162 
163  // update all indexes with new tuple
164  for (const auto& cur : indices) {
165  cur.second->insert(newTuple);
166  }
167 
168  // increment relation size
169  num_tuples++;
170  }

207  /** get index for a given set of keys. Keys are encoded as bits for each column */
208  InterpreterIndex* getIndex(const SearchColumns& key) const {
209  // suffix for order, if no matching prefix exists
210  std::vector<unsigned char> suffix;
211  suffix.reserve(getArity());
212 
213  // convert to order
214  InterpreterIndexOrder order;
215  for (size_t k = 1, i = 0; i < getArity(); i++, k *= 2) {
216  if (key & k) {
217  order.append(i);
218  } else {
219  suffix.push_back(i);
220  }
221  }
222 
223  // see whether there is an order with a matching prefix
224  InterpreterIndex* res = nullptr;
225  {
226  auto lease = lock.acquire();
227  (void)lease;
228  for (auto it = indices.begin(); !res && it != indices.end(); ++it) {
229  if (order.isCompatible(it->first)) {
230  res = it->second.get();
231  }
232  }
233  }
234  // if found, use compatible index
235  if (res) {
236  return res;
237  }
238 
239  // extend index to full index
240  for (auto cur : suffix) {
241  order.append(cur);
242  }
243  assert(order.isComplete());
244 
245  // get a new index
246  return getIndex(order);
247  }
