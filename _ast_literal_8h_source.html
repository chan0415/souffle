56 /**
57  * Subclass of Literal that represents the use of a relation
58  * either in the head or in the body of a Clause, e.g., parent(x,y).
59  * The arguments of the atom can be variables or constants.
60  */
61 class AstAtom : public AstLiteral {
62 protected:
63  /** Name of the atom */
64  AstRelationIdentifier name;
65 
66  /** Arguments of the atom */

68 
69 public:

71

75  const AstRelationIdentifier& getName() const {
76  return name;
77  }

80  size_t getArity() const {
81  return arguments.size();
82  }

85  void setName(const AstRelationIdentifier& n) {
86  name = n;
87  }
88

90  const AstAtom* getAtom() const override {
91  return this;
92  }
93 
94  /** Add argument to the atom */
95  void addArgument(std::unique_ptr<AstArgument> arg) {
96  arguments.push_back(std::move(arg));
97  }
98 
99  /** Return the i-th argument of the atom */
100  AstArgument* getArgument(size_t idx) const {
101  return arguments[idx].get();
102  }
103 
104  /** Replace the argument at the given index with the given argument */
105  void setArgument(size_t idx, std::unique_ptr<AstArgument> newArg) {
106  arguments[idx].swap(newArg);
107  }
108 
109  /** Provides access to the list of arguments of this atom */
110  std::vector<AstArgument*> getArguments() const {
111  return toPtrVector(arguments);
112  }
113 
114  /** Return the number of arguments */
115  size_t argSize() const {
116  return arguments.size();
117  }
118 
119  /** Output to a given stream */
120  void print(std::ostream& os) const override {
121  os << getName() << "(";
122 
123  for (size_t i = 0; i < arguments.size(); ++i) {
124  if (i != 0) {
125  os << ",";
126  }
127  if (arguments[i] != nullptr) {
128  arguments[i]->print(os);
129  } else {
130  os << "_";
131  }
132  }
133  os << ")";
134  }
135 
136  /** Creates a clone if this AST sub-structure */
137  AstAtom* clone() const override {
138  auto res = new AstAtom(name);
139  res->setSrcLoc(getSrcLoc());
140  for (const auto& cur : arguments) {
141  res->arguments.push_back(std::unique_ptr<AstArgument>(cur->clone()));
142  }
143  return res;
144  }
145 
146  /** Mutates this node */
147  void apply(const AstNodeMapper& map) override {
148  for (auto& arg : arguments) {
149  arg = map(std::move(arg));
150  }
151  }
152 
153  /** Obtains a list of all embedded child nodes */
154  std::vector<const AstNode*> getChildNodes() const override {
155  std::vector<const AstNode*> res;
156  for (auto& cur : arguments) {
157  res.push_back(cur.get());
158  }
159  return res;
160  }
161 
162 protected:
163  /** Implements the node comparison for this node type */
164  bool equal(const AstNode& node) const override {
165  assert(dynamic_cast<const AstAtom*>(&node));
166  const AstAtom& other = static_cast<const AstAtom&>(node);
167  return name == other.name && equal_targets(arguments, other.arguments);
168  }
169 };
