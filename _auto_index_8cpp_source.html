29 /** map the keys in the key set to lexicographical order */
30 void AutoIndex::solve() {
31  if (searches.empty()) {
32  return;
33  }
34 
35  // check whether one of the naive indexers should be used
36  // two conditions: either set by environment or relation is a hash map
37  static const char ENV_NAIVE_INDEX[] = "SOUFFLE_USE_NAIVE_INDEX";
38  if (isHashmap || std::getenv(ENV_NAIVE_INDEX)) {
39  static bool first = true;
40 
41  // print a warning - only the first time
42  if (!isHashmap && first) {
43  std::cout << "WARNING: auto index selection disabled, naive indexes are utilized!!\n";
44  first = false;
45  }
46 
47  // every search pattern gets its naive index
48  for (SearchColumns cur : searches) {
49  // obtain order
50  LexicographicalOrder order;
51  SearchColumns mask = cur;
52  for (int i = 0; mask != 0; i++) {
53  if (!(1 << i & mask)) {
54  continue;
55  }
56  order.push_back(i);
57  // clear bit
58  mask &= ~(1 << i);
59  }
60 
61  // add new order
62  orders.push_back(order);
63 
64  // register pseudo chain
65  chainToOrder.push_back(Chain());
66  chainToOrder.back().insert(cur);
67  }
68 
69  // std::cout << "Orders: " << orders << "\n";
70  // std::cout << "Chains: " << chainToOrder << "\n";
71 
72  return;
73  }
74 
75  // Construct the matching poblem
76  for (SearchSet::const_iterator it = searches.begin(); it != searches.end(); ++it) {
77  // For this node check if other nodes are strict subsets
78  for (SearchSet::const_iterator itt = searches.begin(); itt != searches.end(); ++itt) {
79  if (isStrictSubset(*it, *itt)) {
80  matching.addEdge(*it, toB(*itt));
81  }
82  }
83  }
84 
85  // Perform the Hopcroft-Karp on the graph and receive matchings (mapped A->B and B->A)
86  // Assume: alg.calculate is not called on an empty graph
87  ASSERT(!searches.empty());
88  const MaxMatching::Matchings& matchings = matching.calculate();
89 
90  // Extract the chains given the nodes and matchings
91  const ChainOrderMap chains = getChainsFromMatching(matchings, searches);
92 
93  // Should never get no chains back as we never call calculate on an empty graph
94  ASSERT(!chains.empty());
95 
96  for (ChainOrderMap::const_iterator it = chains.begin(); it != chains.end(); ++it) {
97  std::vector<int> ids;
98  SearchColumns initDelta = *(it->begin());
99  insertIndex(ids, initDelta);
100 
101  for (Chain::iterator iit = it->begin(); next(iit) != it->end(); ++iit) {
102  SearchColumns delta = *(next(iit)) - *iit;
103  insertIndex(ids, delta);
104  }
105 
106  ASSERT(!ids.empty());
107 
108  orders.push_back(ids);
109  }
110 
111  // Construct the matching poblem
112  for (SearchSet::const_iterator it = searches.begin(); it != searches.end(); ++it) {
113  int idx = map(*it);
114  size_t l = card(*it);
115  SearchColumns k = 0;
116  for (size_t i = 0; i < l; i++) {
117  k = k + (1 << (orders[idx][i]));
118  }
119  ASSERT(k == *it && "incorrect lexicographical order");
120  }
121 }

123 /** given an unmapped node from set A we follow it from set B until it cannot be matched from B
124  if not mateched from B then umn is a chain*/
125 AutoIndex::Chain AutoIndex::getChain(const SearchColumns umn, const MaxMatching::Matchings& match) {
126  SearchColumns start = umn; // start at an unmateched node
127  Chain chain;
128  // Assume : no circular mappings, i.e. a in A -> b in B -> ........ -> a in A is not allowed.
129  // Given this, the loop will terminate
130  while (true) {
131  MaxMatching::Matchings::const_iterator mit = match.find(toB(start)); // we start from B side
132  chain.insert(start);
133 
134  if (mit == match.end()) {
135  return chain;
136  }
137 
138  SearchColumns a = mit->second;
139  chain.insert(a);
140  start = a;
141  }
142 }

145 const AutoIndex::ChainOrderMap AutoIndex::getChainsFromMatching(
146  const MaxMatching::Matchings& match, const SearchSet& nodes) {
147  ASSERT(!nodes.empty());
148 
149  // Get all unmatched nodes from A
150  const SearchSet& umKeys = getUnmatchedKeys(match, nodes);
151 
152  // Case: if no unmatched nodes then we have an anti-chain
153  if (umKeys.empty()) {
154  for (SearchSet::const_iterator nit = nodes.begin(); nit != nodes.end(); ++nit) {
155  std::set<SearchColumns> a;
156  a.insert(*nit);
157  chainToOrder.push_back(a);
158  return chainToOrder;
159  }
160  }
161 
162  ASSERT(!umKeys.empty());
163 
164  // A worklist of used nodes
165  SearchSet usedKeys;
166 
167  // Case: nodes < umKeys or if nodes == umKeys then anti chain - this is handled by this loop
168  for (SearchSet::iterator it = umKeys.begin(); it != umKeys.end(); ++it) {
169  Chain c = getChain(*it, match);
170  ASSERT(!c.empty());
171  chainToOrder.push_back(c);
172  }
173 
174  ASSERT(!chainToOrder.empty());
175 
176  return chainToOrder;
177 }
