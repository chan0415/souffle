46  // parse relation, split into relation name and values
47  std::pair<std::string, std::vector<std::string>> parseTuple(std::string str) {
48  std::string relName;
49  std::vector<std::string> args;
50 
51  // regex for matching tuples
52  // values matches numbers or strings enclosed in quotation marks
53  std::regex relRegex(
54  "([a-zA-Z0-9_]*)[[:blank:]]*\\(([[:blank:]]*([0-9]+|\"[^\"]*\")([[:blank:]]*,[[:blank:]]*([0-"
55  "9]+|\"[^\"]*\"))*)?\\)",
56  std::regex_constants::extended);
57  std::smatch relMatch;
58 
59  // first check that format matches correctly
60  // and extract relation name
61  if (!std::regex_match(str, relMatch, relRegex) || relMatch.size() < 3) {
62  return std::make_pair(relName, args);
63  }
64 
65  // set relation name
66  relName = relMatch[1];
67 
68  // extract each argument
69  std::string argsList = relMatch[2];
70  std::smatch argsMatcher;
71  std::regex argRegex("[0-9]+|\"[^\"]*\"", std::regex_constants::extended);
72 
73  while (std::regex_search(argsList, argsMatcher, argRegex)) {
74  // match the start of the arguments
75  std::string currentArg = argsMatcher[0];
76  args.push_back(currentArg);
77 
78  // use the rest of the arguments
79  argsList = argsMatcher.suffix().str();
80  }
81 
82  return std::make_pair(relName, args);
83  }

115  // allow scrolling of provenance tree
116  void scrollTree(int maxx, int maxy) {
117  int x = 0;
118  int y = 0;
119 
120  while (1) {
121  int ch = wgetch(treePad);
122 
123  if (ch == KEY_LEFT) {
124  if (x > 2) x -= 3;
125  } else if (ch == KEY_RIGHT) {
126  if (x < MAX_TREE_WIDTH - 3) x += 3;
127  } else if (ch == KEY_UP) {
128  if (y > 0) y -= 1;
129  } else if (ch == KEY_DOWN) {
130  if (y < MAX_TREE_HEIGHT - 1) y += 1;
131  } else {
132  break;
133  }
134 
135  prefresh(treePad, y, x, 0, 0, maxy - 3, maxx - 1);

156  void explain() {

163  char buf[100];

174 
175  // get next command
176  wgetnstr(queryWindow, buf, 100);
177  noecho();
178  curs_set(0);
179  line = buf;
180 
181  // reset tree display on each loop
182  werase(treePad);
183  prefresh(treePad, 0, 0, 0, 0, maxy - 3, maxx - 1);
184  } else {
185  std::cout << "Enter command > ";
186  getline(std::cin, line);
187  }
188 
189  std::vector<std::string> command = split(line, ' ', 1);
190 
191  if (command.size() == 0) {
192  continue;
193  }
194 
195  if (command[0] == "setdepth") {
196  try {
197  depthLimit = std::stoi(command[1]);
198  } catch (std::exception& e) {
199  printStr("Usage: setdepth <depth>\n");
200  continue;
201  }
202  printStr("Depth is now " + std::to_string(depthLimit) + "\n");
203  } else if (command[0] == "explain") {
204  std::pair<std::string, std::vector<std::string>> query;
205  if (command.size() == 2) {
206  query = parseTuple(command[1]);
207  } else {
208  printStr("Usage: explain relation_name(\"<string element1>\", <number element2>, ...)\n");
209  continue;
210  }
211  std::unique_ptr<TreeNode> t = prov.explain(query.first, query.second, depthLimit);
212  printTree(std::move(t));
213  } else if (command[0] == "subproof") {
214  std::pair<std::string, std::vector<std::string>> query;
215  int label = -1;
216  if (command.size() > 1) {
217  query = parseTuple(command[1]);
218  label = std::stoi(query.second[0]);
219  } else {
220  printStr("Usage: subproof relation_name(<label>)\n");
221  continue;
222  }
223  std::unique_ptr<TreeNode> t = prov.explainSubproof(query.first, label, depthLimit);
224  printTree(std::move(t));
225  } else if (command[0] == "rule") {
226  try {
227  auto query = split(command[1], ' ');
228  printStr(prov.getRule(query[0], std::stoi(query[1])) + "\n");
229  } catch (std::exception& e) {
230  printStr("Usage: rule <rule number>\n");
231  continue;
232  }
233  } else if (command[0] == "printrel") {
234  try {
235  printStr(prov.getRelationOutput(command[1]));
236  } catch (std::exception& e) {
237  printStr("Usage: printrel <relation name>\n");
238  continue;
239  }
240  } else if (command[0] == "help") {
241  printStr(
242  "\n----------\n"
243  "Commands:\n"
244  "----------\n"
245  "setdepth <depth>: Set a limit for printed derivation tree height\n"
246  "explain <relation>(<element1>, <element2>, ...): Prints derivation tree\n"
247  "subproof <relation>(<label>): Prints derivation tree for a subproof, label is "

249  "rule <rule number>: Prints a rule\n"

252  } else if (command[0] == "exit") {
253  printStr("Exiting explain\n");

255  }
256 
257  // refresh treePad and allow scrolling

259  prefresh(treePad, 0, 0, 0, 0, maxy - 3, maxx - 1);
260  scrollTree(maxx, maxy);
261  }
262  }

265  }
