104 private:

106 
107  /** Map from node number to SCC number */
108  std::map<const AstRelation*, int> relationToScc;
109

111  std::vector<std::set<unsigned>> successors;
112 
113  /** Predecessor set for the SCC graph */
114  std::vector<std::set<unsigned>> predecessors;
115 
116  /** Relations contained in a SCC */
117  std::vector<std::set<const AstRelation*>> sccToRelation;
118 
119  /** Recursive scR method for computing SCC */
120  void scR(const AstRelation* relation, std::map<const AstRelation*, int>& preOrder, unsigned& counter,
121  std::stack<const AstRelation*>& S, std::stack<const AstRelation*>& P, unsigned& numSCCs);
122 
123  void printDot(std::ostream& os) const;
124 
125  void printJson(std::ostream& os) const;
126 
127 public:
128  static constexpr const char* name = "scc-graph";
129 
130  void run(const AstTranslationUnit& translationUnit) override;
131 
132  /** Return the number of strongly connected components in the SCC graph */
133  size_t size() const {
134  return sccToRelation.size();
135  }
136 
137  /** Get all successor SCCs of a specified scc. */
138  const std::set<unsigned>& successorSCCs(const unsigned scc) const {
139  return successors.at(scc);
140  }
141 
142  /** Get all predecessor SCCs of a specified scc. */
143  const std::set<unsigned>& predecessorSCCs(const unsigned scc) const {
144  return predecessors.at(scc);
145  }
146 
147  /** Get the relations for a given scc. */
148  const std::set<const AstRelation*> relations(const unsigned scc) const {
149  return sccToRelation.at(scc);
150  }
151 
152  /** Get the scc of a given relation. */
153  unsigned scc(const AstRelation* relation) const {
154  return relationToScc.at(relation);
155  }
156 
157  /** Get all inbound and all input relations in the given SCC. */
158  std::set<const AstRelation*> getIns(const unsigned scc) const {
159  auto inbound = getInbound(scc);
160  const auto inputs = getInputs(scc);
161  inbound.insert(inputs.begin(), inputs.end());
162  return inbound;
163  }
164 
165  /** Get all outbound and all output relations in the given SCC. */
166  std::set<const AstRelation*> getOuts(const unsigned scc) const {
167  auto outbound = getOutbound(scc);
168  const auto outputs = getOutputs(scc);
169  outbound.insert(outputs.begin(), outputs.end());
170  return outbound;
171  }
172 
173  /** Get all input relations in the given SCC. */
174  std::set<const AstRelation*> getInputs(const unsigned scc) const {
175  std::set<const AstRelation*> inputs;
176  for (const auto& rel : relations(scc)) {
177  if (rel->isInput()) {
178  inputs.insert(rel);
179  }
180  }
181  return inputs;
182  }
183 
184  /** Get all output relations in the given SCC. */
185  std::set<const AstRelation*> getOutputs(const unsigned scc) const {
186  std::set<const AstRelation*> outputs;
187  for (const auto& rel : relations(scc)) {
188  if (rel->isOutput()) {
189  outputs.insert(rel);
190  }
191  }
192  return outputs;
193  }
194 
195  /** Get all relations in a predecessor SCC that have a successor relation in the given SCC. */
196  std::set<const AstRelation*> getInbound(const unsigned scc) const {
197  std::set<const AstRelation*> inbound;
198  for (const auto& rel : relations(scc)) {
199  for (const auto& pred : precedenceGraph->graph().predecessors(rel)) {
200  if ((unsigned)relationToScc.at(pred) != scc) {
201  inbound.insert(pred);
202  }
203  }
204  }
205  return inbound;
206  }
207 
208  /** Get all relations in the given SCC that have a successor relation in another SCC. */
209  std::set<const AstRelation*> getOutbound(const unsigned scc) const {
210  std::set<const AstRelation*> outbound;
211  for (const auto& rel : relations(scc)) {
212  for (const auto& succ : precedenceGraph->graph().successors(rel)) {
213  if ((unsigned)relationToScc.at(succ) != scc) {
214  outbound.insert(rel);
215  break;
216  }
217  }
218  }
219  return outbound;
220  }
221 
222  /** True if both given relations are in the same SCC, false otherwise. **/
223  bool isInSameSCC(const AstRelation* relation1, const AstRelation* relation2) const {
224  return relationToScc.at(relation1) == relationToScc.at(relation2);
225  }
226 
227  bool isRecursive(const unsigned scc) const {
228  const std::set<const AstRelation*>& sccRelations = sccToRelation.at(scc);
229  if (sccRelations.size() == 1) {
230  const AstRelation* singleRelation = *sccRelations.begin();
231  if (!precedenceGraph->graph().predecessors(singleRelation).count(singleRelation)) {
232  return false;
233  }
234  }
235  return true;
236  }
237 
238  bool isRecursive(const AstRelation* relation) const {
239  return isRecursive(scc(relation));
240  }
241 
242  void print(std::ostream& os, const std::string& format) const;
243 
244  /** Output strongly connected component graph in graphviz format */
245  void print(std::ostream& os) const override;
246 };
