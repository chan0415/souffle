31 class BddbddbConverter : private AstVisitor<void, std::ostream&> {

44 private:
45  /**
46  * The entry point for the conversion of a program, converting the basic top-level structure.
47  */
48  void visitProgram(const AstProgram& program, std::ostream& out) override {
49  // type definition
50  out << "N " << std::numeric_limits<RamDomain>::max() << "\n\n";
51 
52  // variable order
53  int max_attributes = 0;
54  for (const auto& rel : program.getRelations()) {
55  max_attributes = std::max<int>(max_attributes, rel->getAttributes().size());
56  }
57  out << ".bddvarorder ";
58  for (int i = 0; i < max_attributes; i++) {
59  out << "N" << i;
60  if (i + 1 != max_attributes) {
61  out << "_";
62  }
63  }
64  out << "\n\n";
65 
66  // declarations
67  for (const auto& rel : program.getRelations()) {
68  // process the relation declaration
69  visit(*rel, out);
70  }
71  out << "\n";
72 
73  // rules
74  for (const auto& rel : program.getRelations()) {
75  for (const auto& clause : rel->getClauses()) {
76  visit(*clause, out);
77  }
78  }
79  out << "\n";
80  }

82  /**
83  * Converting a relation by creating its declaration.
84  */
85  void visitRelation(const AstRelation& rel, std::ostream& out) override {
86  visitRelationIdentifier(rel.getName(), out);
87 
88  // make nullary relations single-element relations
89  out << "(";
90  if (rel.getAttributes().empty()) {
91  out << "dummy:N0";
92  }
93  int i = 0;
94  out << join(rel.getAttributes(), ",", [&](std::ostream& out, AstAttribute* cur) {
95  out << cur->getAttributeName() << ":N" << (i++);
96  });
97  out << ")";
98 
99  if (rel.isInput()) {
100  out << " inputtuples";
101  }
102  if (rel.isOutput()) {
103  out << " outputtuples";
104  }
105 
106  out << "\n";
107  }
108 
109  /**
110  * Converting a clause.
111  */
112  void visitClause(const AstClause& clause, std::ostream& out) override {
113  visit(*clause.getHead(), out);
114 
115  // if it is a fact, that's it
116  if (clause.isFact()) {
117  // there must not be any additons
118  if (!extra_literals.empty()) {
119  throw UnsupportedConstructException("Unsupported fact: " + toString(clause));
120  }
121  out << ".\n";
122  return;
123  }
124 
125  // convert the body
126  out << " :- ";
127  out << join(
128  clause.getBodyLiterals(), ",", [&](std::ostream& out, AstLiteral* cur) { visit(*cur, out); });
129 
130  // add extra_literals
131  for (const auto& cur : extra_literals) {
132  out << "," << cur;
133  }
134  extra_literals.clear();
135 
136  // done
137  out << ".\n";
138  }

140  void visitAtom(const AstAtom& atom, std::ostream& out) override {
141  visitRelationIdentifier(atom.getName(), out);
142 
143  // since no nullary-relations are allowed, we add a dummy value
144  if (atom.getArguments().empty()) {
145  out << "(0)";

148 
149  out << "(";

151  out << ")";
152  }

178  void visitFunctor(const AstFunctor& fun, std::ostream& out) override {
179  // create a new variable
180  std::string var = "aux_var_" + toString(varCounter++);
181 
182  // write variable
183  out << var;
184 
185  // assign variable in extra clause
186  std::stringstream binding;
187  binding << var << "=";
188 
189  // process unary operators
190  if (dynamic_cast<const AstUnaryFunctor*>(&fun)) {
191  // binary functors are not supported
192  throw UnsupportedConstructException("Unsupported function: " + toString(fun));
193  } else if (const AstBinaryFunctor* binary = dynamic_cast<const AstBinaryFunctor*>(&fun)) {
194  visit(*binary->getLHS(), binding);
195  binding << getSymbolForBinaryOp(binary->getFunction());
196  visit(*binary->getRHS(), binding);
197  } else {
198  assert(false && "Unsupported functor!");
199  }
200  extra_literals.push_back(binding.str());
201  }
