119 class AstExecutionPlan : public AstNode {
120 public:
121  typedef typename std::map<int, AstExecutionOrder>::const_iterator const_iterator;
122 
123 private:
124  /** Mapping versions of clauses to execution plans */
125  std::map<int, std::unique_ptr<AstExecutionOrder>> plans;
126 
127  /** remember maximal version number */
128  int maxVersion;
129 
130 public:
131  AstExecutionPlan() : maxVersion(-1) {}
132 
133  /** Updates the execution order for a special version of a rule */
134  void setOrderFor(int version, std::unique_ptr<AstExecutionOrder> plan) {
135  plans[version] = std::move(plan);
136  if (version > maxVersion) {
137  maxVersion = version;
138  }
139  }
140 
141  /** Determines whether for the given version a plan has been specified */
142  bool hasOrderFor(int version) const {
143  return plans.find(version) != plans.end();
144  }
145 
146  /** get maximal version number */
147  const int getMaxVersion() const {
148  return maxVersion;
149  }
150 
151  /** Obtains the order defined for the given version */
152  const AstExecutionOrder& getOrderFor(int version) const {
153  assert(hasOrderFor(version));
154  return *plans.find(version)->second;
155  }
156 
157  /** Tests whether there has any order been defined */
158  bool empty() const {
159  return plans.empty();
160  }
161 
162  std::map<int, const AstExecutionOrder*> getOrders() const {
163  std::map<int, const AstExecutionOrder*> result;
164  for (auto& plan : plans) {
165  result.insert(std::make_pair(plan.first, plan.second.get()));
166  }
167  return result;
168  }
169 
170  void print(std::ostream& out) const override {
171  if (!plans.empty()) {
172  out << "\n\n .plan ";
173  bool first = true;
174  for (auto it = plans.begin(); it != plans.end(); ++it) {
175  if (first) {
176  first = false;
177  } else {
178  out << ",";
179  }
180  out << (*it).first << ":";
181  (*it).second->print(out);
182  }
183  }
184  }
185 
186  AstExecutionPlan* clone() const override {
187  auto res = new AstExecutionPlan();
188  res->setSrcLoc(getSrcLoc());
189  for (auto& plan : plans) {
190  res->setOrderFor(plan.first, std::unique_ptr<AstExecutionOrder>(plan.second->clone()));
191  }
192  return res;
193  }
194 
195  void apply(const AstNodeMapper& map) override {
196  for (auto& plan : plans) {
197  plan.second = map(std::move(plan.second));
198  }
199  }
200 
201  /** Obtains a list of all embedded child nodes */
202  std::vector<const AstNode*> getChildNodes() const override {
203  std::vector<const AstNode*> childNodes;
204  for (auto& plan : plans) {
205  childNodes.push_back(plan.second.get());
206  }
207  return childNodes;
208  }
209 
210 protected:
211  /** Implements the node comparison for this node type */
212  bool equal(const AstNode& node) const override {
213  assert(dynamic_cast<const AstExecutionPlan*>(&node));
214  const AstExecutionPlan& other = static_cast<const AstExecutionPlan&>(node);
215  if (maxVersion != other.maxVersion) {
216  return false;
217  }
218  if (plans.size() != other.plans.size()) {
219  return false;
220  }
221  auto iter = plans.begin();
222  auto otherIter = other.plans.begin();
223  for (; iter != plans.end(); ++iter, ++otherIter) {
224  if (iter->first != otherIter->first) {
225  return false;
226  }
227  if (*iter->second != *otherIter->second) {
228  return false;
229  }
230  }
231  return true;
232  }

235 /**
236  * Intermediate representation of a datalog clause.
237  *
238  * A clause can either be:
239  * - a fact - a clause with no body (e.g., X(a,b))
240  * - a rule - a clause with a head and a body (e.g., Y(a,b) -: X(a,b))
241  *
242  * TODO: Currently Clause object is used to represent 2 different types of datalog
243  * clauses, such as rules, queries and facts. This solution was to quickly
244  * overcome issues related to bottom-up construction of IR. In future,
245  * Clause should be made abstract and have 2 subclasses: Rule and Fact.
