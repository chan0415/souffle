445  void visitScan(const RamScan& scan) override {
446  // get the targeted relation
447  const InterpreterRelation& rel = env.getRelation(scan.getRelation());
448 
449  // process full scan if no index is given
450  if (scan.getRangeQueryColumns() == 0) {
451  // if scan is not binding anything => check for emptiness
452  if (scan.isPureExistenceCheck() && !rel.empty()) {
453  visitSearch(scan);
454  return;
455  }
456 
457  // if scan is unrestricted => use simple iterator
458  for (const RamDomain* cur : rel) {
459  ctxt[scan.getLevel()] = cur;
460  visitSearch(scan);
461  }
462  return;
463  }
464 
465  // create pattern tuple for range query
466  auto arity = rel.getArity();
467  RamDomain low[arity];
468  RamDomain hig[arity];
469  auto pattern = scan.getRangePattern();
470  for (size_t i = 0; i < arity; i++) {
471  if (pattern[i] != nullptr) {
472  low[i] = eval(pattern[i], env, ctxt);
473  hig[i] = low[i];
474  } else {
475  low[i] = MIN_RAM_DOMAIN;
476  hig[i] = MAX_RAM_DOMAIN;
477  }
478  }
479 
480  // obtain index
481  auto idx = rel.getIndex(scan.getRangeQueryColumns(), nullptr);
482 
483  // get iterator range
484  auto range = idx->lowerUpperBound(low, hig);
485 
486  // if this scan is not binding anything ...
487  if (scan.isPureExistenceCheck()) {
488  if (range.first != range.second) {
489  visitSearch(scan);
490  }
491  return;
492  }
493 
494  // conduct range query
495  for (auto ip = range.first; ip != range.second; ++ip) {
496  const RamDomain* data = *(ip);
497  ctxt[scan.getLevel()] = data;
498  visitSearch(scan);
499  }

522  void visitAggregate(const RamAggregate& aggregate) override {
523  // get the targeted relation
524  const InterpreterRelation& rel = env.getRelation(aggregate.getRelation());
525 
526  // initialize result
527  RamDomain res = 0;
528  switch (aggregate.getFunction()) {
529  case RamAggregate::MIN:
530  res = MAX_RAM_DOMAIN;
531  break;
532  case RamAggregate::MAX:
533  res = MIN_RAM_DOMAIN;
534  break;
535  case RamAggregate::COUNT:
536  res = 0;
537  break;
538  case RamAggregate::SUM:
539  res = 0;
540  break;
541  }
542 
543  // init temporary tuple for this level
544  auto arity = rel.getArity();
545 
546  // get lower and upper boundaries for iteration
547  const auto& pattern = aggregate.getPattern();
548  RamDomain low[arity];
549  RamDomain hig[arity];
550 
551  for (size_t i = 0; i < arity; i++) {
552  if (pattern[i] != nullptr) {
553  low[i] = eval(pattern[i], env, ctxt);
554  hig[i] = low[i];
555  } else {
556  low[i] = MIN_RAM_DOMAIN;
557  hig[i] = MAX_RAM_DOMAIN;
558  }
559  }
560 
561  // obtain index
562  auto idx = rel.getIndex(aggregate.getRangeQueryColumns());
563 
564  // get iterator range
565  auto range = idx->lowerUpperBound(low, hig);
566 
567  // check for emptiness
568  if (aggregate.getFunction() != RamAggregate::COUNT) {
569  if (range.first == range.second) {
570  return; // no elements => no min/max
571  }
572  }
573 
574  // iterate through values
575  for (auto ip = range.first; ip != range.second; ++ip) {
576  // link tuple
577  const RamDomain* data = *(ip);
578  ctxt[aggregate.getLevel()] = data;
579 
580  // count is easy
581  if (aggregate.getFunction() == RamAggregate::COUNT) {
582  res++;
583  continue;
584  }
585 
586  // aggregation is a bit more difficult
587 
588  // eval target expression
589  RamDomain cur = eval(aggregate.getTargetExpression(), env, ctxt);
590 
591  switch (aggregate.getFunction()) {
592  case RamAggregate::MIN:
593  res = std::min(res, cur);
594  break;
595  case RamAggregate::MAX:
596  res = std::max(res, cur);
597  break;
598  case RamAggregate::COUNT:
599  res = 0;
600  break;
601  case RamAggregate::SUM:
602  res += cur;
603  break;
604  }
605  }
606 
607  // write result to environment
608  RamDomain tuple[1];
609  tuple[0] = res;
610  ctxt[aggregate.getLevel()] = tuple;
611 
612  // check whether result is used in a condition
613  auto condition = aggregate.getCondition();
614  if (condition && !eval(*condition, env, ctxt)) {
615  return; // condition not valid => skip nested
616  }
617 
618  // run nested part - using base class visitor
619  visitSearch(aggregate);
620  }

893 Order scheduleByModel(AstClause& clause, InterpreterEnvironment& env, std::ostream* report) {
894  assert(!clause.isFact());
895 
896  // check whether schedule is fixed
897  if (clause.hasFixedExecutionPlan()) {
898  if (report) {
899  *report << " Skipped due to fixed execution plan!\n";
900  }
901  return Order::getIdentity(clause.getAtoms().size());
902  }
903 
904  // check whether there is actually something to schedule
905  if (clause.getAtoms().size() < 2) {
906  return Order::getIdentity(clause.getAtoms().size());
907  }
908 
909  // TODO: provide alternative scheduling approach for larger rules
910  // 8 atoms require ~200ms to schedule
911  // 9 atoms require ~2400ms to schedule
912  // 10 atoms require ~29000ms to schedule
913  // 11 atoms => out of memory
914  if (clause.getAtoms().size() > 8) {
915  return Order::getIdentity(clause.getAtoms().size());
916  }
917 
918  // get atom list
919  std::vector<AstAtom*> atoms = clause.getAtoms();
920 
921  // a utility for mapping variable names to ids
922  std::map<std::string, int> varIDs;
923  auto getID = [&](const AstVariable& var) -> int {
924  auto pos = varIDs.find(var.getName());
925  if (pos != varIDs.end()) {
926  return pos->second;
927  }
928  int id = varIDs.size();
929  varIDs[var.getName()] = id;
930  return id;
931  };
932 
933  // fix scheduling strategy
934  typedef Problem<SimpleComputationalCostModel> Problem;
935  typedef typename Problem::atom_type Atom;
936 
937  // create an optimization problem
938  Problem p;
939 
940  // create atoms
941  for (unsigned i = 0; i < atoms.size(); i++) {
942  // convert pattern of arguments
943  std::vector<Argument> args;
944 
945  for (const AstArgument* arg : atoms[i]->getArguments()) {
946  if (const AstVariable* var = dynamic_cast<const AstVariable*>(arg)) {
947  args.push_back(Argument::createVar(getID(*var)));
948  } else if (dynamic_cast<const AstUnnamedVariable*>(arg)) {
949  args.push_back(Argument::createUnderscore());
950  } else if (dynamic_cast<const AstConstant*>(arg)) {
951  args.push_back(Argument::createConst());
952  } else {
953  args.push_back(Argument::createOther());
954  }
955  }
956 
957  // add new atom
958  AstTranslator translator;
959  p.addAtom(
960  Atom(i, args, env.getRelation(translator.translateRelationName(atoms[i]->getName())).size()));
961  }
962 
963  // solve the optimization problem
964  auto schedule = p.solve();
965 
966  // log problem and solution
967  if (report) {
968  *report << "Scheduling Problem: " << p << "\n";
969  *report << " Schedule: " << schedule << "\n";
970  }
971 
972  // extract order
973  Order res;
974  for (const auto& cur : schedule) {
975  res.append(cur.getID());
976  }
977 
978  // re-order atoms
979  clause.reorderAtoms(res.getOrder());
980 
981  // done
982  return res;
983 }

1051 RelationStats RelationStats::extractFrom(const InterpreterRelation& rel, uint32_t sample_size) {
1052  // write each column in its own set
1053  std::vector<btree_set<RamDomain>> columns(rel.getArity());
1054 
1055  // analyze sample
1056  uint32_t count = 0;
1057  for (auto it = rel.begin(); it != rel.end() && count < sample_size; ++it, ++count) {
1058  const RamDomain* tuple = *it;
1059 
1060  // compute cardinality of columns
1061  for (std::size_t i = 0; i < rel.getArity(); ++i) {
1062  columns[i].insert(tuple[i]);
1063  }
1064  }
1065 
1066  // create the resulting statistics object
1067  RelationStats stats;
1068 
1069  stats.arity = rel.getArity();
1070  stats.size = rel.size();
1071  stats.sample_size = count;
1072 
1073  for (std::size_t i = 0; i < rel.getArity(); i++) {
1074  // estimate the cardinality of the columns
1075  uint64_t card = 0;
1076  if (count > 0) {
1077  // based on the observed probability
1078  uint64_t cur = columns[i].size();
1079  double p = ((double)cur / (double)count);
1080 
1081  // obtain an estimate of the overall cardinality
1082  card = (uint64_t)(p * rel.size());
1083 
1084  // make sure that it is at least what you have seen
1085  if (card < cur) {
1086  card = cur;
1087  }
1088  }
1089 
1090  // add result
1091  stats.cardinalities.push_back(card);
1092  }
1093 
1094  // done
1095  return stats;
1096 }
