43 void RuleBody::conjunct(RuleBody&& other) {
44  std::vector<clause> res;
45 
46  for (const auto& clauseA : dnf) {
47  for (const auto& clauseB : other.dnf) {
48  // create a new clause in result
49  clause cur;
50 
51  // it is the concatenation of the two clauses
52  for (const auto& lit : clauseA) {
53  cur.emplace_back(lit.clone());
54  }
55  for (const auto& lit : clauseB) {
56  insert(cur, lit.clone());
57  }

60  }
61  }
62 
63  // update local dnf
64  dnf.swap(res);

75 std::vector<AstClause*> RuleBody::toClauseBodies() const {
76  // collect clause results
77  std::vector<AstClause*> bodies;
78  for (const clause& cur : dnf) {
79  bodies.push_back(new AstClause());
80  AstClause& clause = *bodies.back();
81 
82  for (const literal& lit : cur) {
83  // extract literal
84  AstLiteral* base = lit.atom->clone();
85  // negate if necessary
86  if (lit.negated) {
87  // negate
88  if (AstAtom* atom = dynamic_cast<AstAtom*>(base)) {
89  base = new AstNegation(std::unique_ptr<AstAtom>(atom));
90  base->setSrcLoc(atom->getSrcLoc());
91  } else if (AstConstraint* cstr = dynamic_cast<AstConstraint*>(base)) {
92  cstr->negate();
93  }
94  }
95 
96  // add to result
97  clause.addToBody(std::unique_ptr<AstLiteral>(base));
98  }
99  }
100 
101  // done
102  return bodies;
103 }

195 void RuleBody::insert(std::vector<clause>& cnf, clause&& cls) {
196  for (const auto& cur : cnf) {
197  if (isSubsetOf(cur, cls)) {
198  return;
199  }
200  }
201  std::vector<clause> res;
202  for (auto& cur : cnf) {
203  if (!isSubsetOf(cls, cur)) {
204  res.emplace_back(std::move(cur));
205  }
206  }
207  res.swap(cnf);
208  cnf.emplace_back(std::move(cls));
209 }
