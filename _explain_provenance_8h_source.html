31 inline std::vector<std::string> split(std::string s, char delim, int times = -1) {
32  std::vector<std::string> v;
33  std::stringstream ss(s);
34  std::string item;
35 
36  while ((times > 0 || times <= -1) && std::getline(ss, item, delim)) {
37  v.push_back(item);
38  times--;
39  }
40 
41  if (ss.peek() != EOF) {
42  std::string remainder;
43  std::getline(ss, remainder);
44  v.push_back(remainder);
45  }
46 
47  return v;
48 }
