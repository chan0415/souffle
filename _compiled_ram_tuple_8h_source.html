28 /**
29  * The type of object stored within relations representing the actual
30  * tuple value. Each tuple consists of a constant number of components.
31  *
32  * @tparam Domain the domain of the component values
33  * @tparam arity the number of components within an instance
34  */
35 template <typename Domain, std::size_t _arity>
36 struct Tuple {
37  // some features for template meta programming
38  typedef Domain value_type;
39  enum { arity = _arity };
40 
41  // the stored data
42  Domain data[arity];
43 
44  // constructores, destructors and assignment are default
45 
46  // provide access to components
47  const Domain& operator[](std::size_t index) const {
48  return data[index];
49  }
50 
51  // provide access to components
52  Domain& operator[](std::size_t index) {
53  return data[index];
54  }
55 
56  // a comparison operation
57  bool operator==(const Tuple& other) const {
58  for (std::size_t i = 0; i < arity; i++) {
59  if (data[i] != other.data[i]) return false;
60  }
61  return true;
62  }
63 
64  // inequality comparison
65  bool operator!=(const Tuple& other) const {
66  return !(*this == other);
67  }
68 
69  // required to put tuples into e.g. a std::set container
70  bool operator<(const Tuple& other) const {
71  for (std::size_t i = 0; i < arity; ++i) {
72  if (data[i] < other.data[i]) return true;
73  if (data[i] > other.data[i]) return false;
74  }
75  return false;
76  }
77 
78  // required to put tuples into e.g. a btree container
79  bool operator>(const Tuple& other) const {
80  for (std::size_t i = 0; i < arity; ++i) {
81  if (data[i] > other.data[i]) return true;
82  if (data[i] < other.data[i]) return false;
83  }
84  return false;
85  }
86 
87  // allow tuples to be printed
88  friend std::ostream& operator<<(std::ostream& out, const Tuple& tuple) {
89  if (arity == 0) return out << "[]";
90  out << "[";
91  for (std::size_t i = 0; i < (std::size_t)(arity - 1); ++i) {
92  out << tuple.data[i];
93  out << ",";
94  }
95  return out << tuple.data[arity - 1] << "]";
96  }
