53  /** calculates max matching */

55  while (bfSearch()) {
56  for (Graph::iterator it = graph.begin(); it != graph.end(); ++it) {
57  if (getMatch(it->first) == NIL) {
58  dfSearch(it->first);
59  }
60  }
61  }

63  }

69  /** add an edge to the bi-partite graph */
70  void addEdge(SearchColumns u, SearchColumns v) {
71  if (graph.find(u) == graph.end()) {
72  Edges vals;
73  vals.insert(v);
74  graph.insert(make_pair(u, vals));
