48  /** Convenience method to copy strings between symbol tables and resolve their references. */
49  inline void copyAll() {
50  for (auto& cur : numToStr) {
51  cur = strdup(cur);
52  }
53  for (auto& cur : strToNum) {
54  const_cast<std::string&>(cur.first) = numToStr[cur.second];
55  }
56  }

58  /** Convenience method to free memory allocated for strings. */
59  inline void freeAll() {

62  }
63  }

65  /** Convenience method to place a new symbol in the table, if it does not exist, and return the index of
66  * it. */
67  inline const size_t newSymbolOfIndex(const char* symbol) {
68  size_t idx;
69  auto it = strToNum.find(symbol);
70  if (it == strToNum.end()) {
71  char* str = strdup(symbol);
72  idx = numToStr.size();
73  strToNum[str] = idx;
74  numToStr.push_back(str);
75  } else {
76  idx = it->second;
77  }
78  return idx;
79  }

81  /** Convenience method to place a new symbol in the table, if it does not exist. */
82  inline void newSymbol(const char* symbol) {
83  if (strToNum.find(symbol) == strToNum.end()) {
84  char* str = strdup(symbol);
85  strToNum[str] = numToStr.size();
86  numToStr.push_back(str);
87  }
88  }

176  /** Bulk insert symbols into the table, note that this operation is more efficient than repeated inserts

178  void insert(const char** symbols, const size_t n) {
