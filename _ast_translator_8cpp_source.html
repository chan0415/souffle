379 std::unique_ptr<RamValue> translateValue(const AstArgument* arg, const ValueIndex& index = ValueIndex()) {
380  std::unique_ptr<RamValue> val;
381  if (!arg) {
382  return val;
383  }
384 
385  if (const AstVariable* var = dynamic_cast<const AstVariable*>(arg)) {
386  ASSERT(index.isDefined(*var) && "variable not grounded");
387  const Location& loc = index.getDefinitionPoint(*var);
388  val = std::make_unique<RamElementAccess>(loc.level, loc.component, loc.name);
389  } else if (dynamic_cast<const AstUnnamedVariable*>(arg)) {
390  return nullptr; // utilised to identify _ values
391  } else if (const AstConstant* c = dynamic_cast<const AstConstant*>(arg)) {
392  val = std::make_unique<RamNumber>(c->getIndex());
393  } else if (const AstUnaryFunctor* uf = dynamic_cast<const AstUnaryFunctor*>(arg)) {
394  val = std::unique_ptr<RamValue>(
395  new RamUnaryOperator(uf->getFunction(), translateValue(uf->getOperand(), index)));
396  } else if (const AstBinaryFunctor* bf = dynamic_cast<const AstBinaryFunctor*>(arg)) {
397  val = std::make_unique<RamBinaryOperator>(
398  bf->getFunction(), translateValue(bf->getLHS(), index), translateValue(bf->getRHS(), index));
399  } else if (const AstTernaryFunctor* tf = dynamic_cast<const AstTernaryFunctor*>(arg)) {
400  val = std::unique_ptr<RamValue>(
401  new RamTernaryOperator(tf->getFunction(), translateValue(tf->getArg(0), index),
402  translateValue(tf->getArg(1), index), translateValue(tf->getArg(2), index)));
403  } else if (dynamic_cast<const AstCounter*>(arg)) {
404  val = std::make_unique<RamAutoIncrement>();
405  } else if (const AstRecordInit* init = dynamic_cast<const AstRecordInit*>(arg)) {
406  std::vector<std::unique_ptr<RamValue>> values;
407  for (const auto& cur : init->getArguments()) {
408  values.push_back(translateValue(cur, index));
409  }
410  val = std::make_unique<RamPack>(std::move(values));
411  } else if (const AstAggregator* agg = dynamic_cast<const AstAggregator*>(arg)) {
412  // here we look up the location the aggregation result gets bound
413  auto loc = index.getAggregatorLocation(*agg);
414  val = std::make_unique<RamElementAccess>(loc.level, loc.component, loc.name);
415  } else if (const AstSubroutineArgument* subArg = dynamic_cast<const AstSubroutineArgument*>(arg)) {
416  val = std::make_unique<RamArgument>(subArg->getNumber());
417  } else {
418  std::cout << "Unsupported node type of " << arg << ": " << typeid(*arg).name() << "\n";
419  ASSERT(false && "unknown AST node type not permissible");
420  }
421 
422  return val;
423 }

431 std::unique_ptr<RamStatement> AstTranslator::translateClause(const AstClause& clause,
432  const AstProgram* program, const TypeEnvironment* typeEnv, int version, bool ret) {
433  // check whether there is an imposed order constraint
434  if (clause.getExecutionPlan() && clause.getExecutionPlan()->hasOrderFor(version)) {
435  // get the imposed order
436  const AstExecutionOrder& order = clause.getExecutionPlan()->getOrderFor(version);
437 
438  // create a copy and fix order
439  std::unique_ptr<AstClause> copy(clause.clone());
440 
441  // Change order to start at zero
442  std::vector<unsigned int> newOrder(order.size());
443  std::transform(order.begin(), order.end(), newOrder.begin(),
444  [](unsigned int i) -> unsigned int { return i - 1; });
445 
446  // re-order atoms
447  copy->reorderAtoms(newOrder);
448 
449  // clear other order and fix plan
450  copy->clearExecutionPlan();
451  copy->setFixedExecutionPlan();
452 
453  // translate reordered clause
454  return translateClause(*copy, program, typeEnv, version);
455  }
456 
457  // get extract some details
458  const AstAtom& head = *clause.getHead();
459 
460  // a utility to translate atoms to relations
461  auto getRelation = [&](const AstAtom* atom) {
462  return getRamRelation((program ? getAtomRelation(atom, program) : nullptr), typeEnv,
463  getRelationName(atom->getName()), atom->getArity());
464  };
465 
466  // handle facts
467  if (clause.isFact()) {
468  // translate arguments
469  std::vector<std::unique_ptr<RamValue>> values;
470  for (auto& arg : clause.getHead()->getArguments()) {
471  values.push_back(translateValue(*arg));
472  }
473 
474  // create a fact statement
475  return std::make_unique<RamFact>(getRelation(&head), std::move(values));
476  }
477 
478  // the rest should be rules
479  assert(clause.isRule());
480 
481  // -- index values in rule --
482 
483  // create value index
484  ValueIndex valueIndex;
485 
486  // the order of processed operations
487  std::vector<const AstNode*> op_nesting;
488 
489  int level = 0;
490  for (AstAtom* atom : clause.getAtoms()) {
491  // index nested variables and records
492  typedef std::vector<AstArgument*> arg_list;
493  // std::map<const arg_list*, int> arg_level;
494  std::map<const AstNode*, std::unique_ptr<arg_list>> nodeArgs;
495 
496  std::function<arg_list*(const AstNode*)> getArgList = [&](const AstNode* curNode) {
497  if (!nodeArgs.count(curNode)) {
498  if (auto rec = dynamic_cast<const AstRecordInit*>(curNode)) {
499  nodeArgs.insert(std::make_pair(curNode, std::make_unique<arg_list>(rec->getArguments())));
500  } else if (auto atom = dynamic_cast<const AstAtom*>(curNode)) {
501  nodeArgs.insert(
502  std::make_pair(curNode, std::make_unique<arg_list>(atom->getArguments())));
503  } else {
504  assert(false && "node type doesn't have arguments!");
505  }
506  }
507  arg_list* cur = nodeArgs[curNode].get();
508  return cur;
509  };
510 
511  std::map<const arg_list*, int> arg_level;
512  nodeArgs.insert(std::make_pair(atom, std::make_unique<arg_list>(atom->getArguments())));
513  // the atom is obtained at the current level
514  arg_level[nodeArgs[atom].get()] = level;
515  op_nesting.push_back(atom);
516 
517  // increment nesting level for the atom
518  level++;
519 
520  // relation
521  std::unique_ptr<RamRelation> relation = getRelation(atom);
522 
523  std::function<void(const AstNode*)> indexValues = [&](const AstNode* curNode) {
524  arg_list* cur = getArgList(curNode);
525  for (size_t pos = 0; pos < cur->size(); pos++) {
526  // get argument
527  auto& arg = (*cur)[pos];
528 
529  // check for variable references
530  if (auto var = dynamic_cast<const AstVariable*>(arg)) {
531  if (pos < relation->getArity()) {
532  valueIndex.addVarReference(*var, arg_level[cur], pos, relation->getArg(pos));
533  } else {
534  valueIndex.addVarReference(*var, arg_level[cur], pos);
535  }
536  }
537 
538  // check for nested records
539  if (auto rec = dynamic_cast<const AstRecordInit*>(arg)) {
540  // introduce new nesting level for unpack
541  int unpack_level = level++;
542  op_nesting.push_back(rec);
543  arg_level[getArgList(rec)] = unpack_level;
544  valueIndex.setRecordUnpackLevel(*rec, unpack_level);
545 
546  // register location of record
547  valueIndex.setRecordDefinition(*rec, arg_level[cur], pos);
548 
549  // resolve nested components
550  indexValues(rec);
551  }
552  }
553  };
554 
555  indexValues(atom);
556  }
557 
558  // add aggregation functions
559  std::vector<const AstAggregator*> aggregators;
560  visitDepthFirstPostOrder(clause, [&](const AstAggregator& cur) {
561 
562  // add each aggregator expression only once
563  if (any_of(aggregators, [&](const AstAggregator* agg) { return *agg == cur; })) {
564  return;
565  }
566 
567  int aggLoc = level++;
568  valueIndex.setAggregatorLocation(cur, Location({aggLoc, 0}));
569 
570  // bind aggregator variables to locations
571  assert(dynamic_cast<const AstAtom*>(cur.getBodyLiterals()[0]));
572  const AstAtom& atom = static_cast<const AstAtom&>(*cur.getBodyLiterals()[0]);
573  for (size_t pos = 0; pos < atom.getArguments().size(); ++pos) {
574  if (const AstVariable* var = dynamic_cast<const AstVariable*>(atom.getArgument(pos))) {
575  valueIndex.addVarReference(*var, aggLoc, (int)pos, getRelation(&atom)->getArg(pos));
576  }
577  };
578 
579  // and remember aggregator
580  aggregators.push_back(&cur);
581  });
582 
583  // -- create RAM statement --
584 
585  // begin with projection
586  std::unique_ptr<RamOperation> op;
587  if (ret) {
588  RamReturn* returnValue = new RamReturn(level);
589 
590  // get all values in the body
591  for (AstLiteral* lit : clause.getBodyLiterals()) {
592  if (auto atom = dynamic_cast<AstAtom*>(lit)) {
593  for (AstArgument* arg : atom->getArguments()) {
594  returnValue->addValue(translateValue(arg, valueIndex));
595  }
596  } else if (auto neg = dynamic_cast<AstNegation*>(lit)) {
597  for (size_t i = 0; i < neg->getAtom()->getArguments().size() - 2; i++) {
598  auto arg = neg->getAtom()->getArguments()[i];
599  returnValue->addValue(translateValue(arg, valueIndex));
600  }
601  returnValue->addValue(std::make_unique<RamNumber>(-1));
602  returnValue->addValue(std::make_unique<RamNumber>(-1));
603  }
604  }
605 
606  op = std::unique_ptr<RamOperation>(returnValue);
607  } else {
608  std::unique_ptr<RamProject> project =
609  std::unique_ptr<RamProject>(new RamProject(getRelation(&head), level));
610 
611  for (AstArgument* arg : head.getArguments()) {
612  project->addArg(translateValue(arg, valueIndex));
613  }
614 
615  // check existence for original tuple if we have provenance
616  if (Global::config().has("provenance")) {
617  auto uniquenessEnforcement = std::make_unique<RamNotExists>(getRelation(&head));
618  auto arity = head.getArity() - 2;
619 
620  bool add = true;
621  // add args for original tuple
622  for (size_t i = 0; i < arity; i++) {
623  auto arg = head.getArgument(i);
624 
625  // don't add counters
626  if (dynamic_cast<AstCounter*>(arg)) {
627  add = false;
628  break;
629  }
630 
631  uniquenessEnforcement->addArg(translateValue(arg, valueIndex));
632  }
633 
634  // add two unnamed args for provenance columns
635  uniquenessEnforcement->addArg(nullptr);
636  uniquenessEnforcement->addArg(nullptr);
637 
638  if (add) {
639  project->addCondition(std::move(uniquenessEnforcement), *project);
640  }
641  }

643  // build up insertion call
644  op = std::move(project); // start with innermost
645  }
646 
647  // add aggregator levels
648  for (auto it = aggregators.rbegin(); it != aggregators.rend(); ++it) {
649  const AstAggregator* cur = *it;
650  level--;
651 
652  // translate aggregation function
653  RamAggregate::Function fun = RamAggregate::MIN;
654  switch (cur->getOperator()) {
655  case AstAggregator::min:
656  fun = RamAggregate::MIN;
657  break;
658  case AstAggregator::max:
659  fun = RamAggregate::MAX;
660  break;
661  case AstAggregator::count:
662  fun = RamAggregate::COUNT;
663  break;
664  case AstAggregator::sum:
665  fun = RamAggregate::SUM;
666  break;
667  }
668 
669  // translate target expression
670  std::unique_ptr<RamValue> value = translateValue(cur->getTargetExpression(), valueIndex);
671 
672  // translate body literal
673  assert(cur->getBodyLiterals().size() == 1 && "Unsupported complex aggregation body encountered!");
674  const AstAtom* atom = dynamic_cast<const AstAtom*>(cur->getBodyLiterals()[0]);
675  assert(atom && "Unsupported complex aggregation body encountered!");
676 
677  // add Ram-Aggregation layer
678  op = std::unique_ptr<RamOperation>(
679  new RamAggregate(std::move(op), fun, std::move(value), getRelation(atom)));
680 
681  // add constant constraints
682  for (size_t pos = 0; pos < atom->argSize(); ++pos) {
683  if (AstConstant* c = dynamic_cast<AstConstant*>(atom->getArgument(pos))) {
684  op->addCondition(std::unique_ptr<RamCondition>(new RamBinaryRelation(BinaryConstraintOp::EQ,
685  std::unique_ptr<RamValue>(
686  new RamElementAccess(level, pos, getRelation(atom)->getArg(pos))),
687  std::make_unique<RamNumber>(c->getIndex()))));
688  }
689  }
690  }
691 
692  // build operation bottom-up
693  while (!op_nesting.empty()) {
694  // get next operator
695  const AstNode* cur = op_nesting.back();
696  op_nesting.pop_back();
697 
698  // get current nesting level
699  auto level = op_nesting.size();
700 
701  if (const AstAtom* atom = dynamic_cast<const AstAtom*>(cur)) {
702  // find out whether a "search" or "if" should be issued
703  bool isExistCheck = !valueIndex.isSomethingDefinedOn(level);
704  for (size_t pos = 0; pos < atom->argSize(); ++pos) {
705  if (dynamic_cast<AstAggregator*>(atom->getArgument(pos))) {
706  isExistCheck = false;
707  }
708  }
709 
710  // add a scan level
711  op = std::make_unique<RamScan>(getRelation(atom), std::move(op), isExistCheck);
712 
713  // add constraints
714  for (size_t pos = 0; pos < atom->argSize(); ++pos) {
715  if (AstConstant* c = dynamic_cast<AstConstant*>(atom->getArgument(pos))) {
716  op->addCondition(std::unique_ptr<RamCondition>(new RamBinaryRelation(
717  BinaryConstraintOp::EQ, std::unique_ptr<RamValue>(new RamElementAccess(
718  level, pos, getRelation(atom)->getArg(pos))),
719  std::make_unique<RamNumber>(c->getIndex()))));
720  } else if (AstAggregator* agg = dynamic_cast<AstAggregator*>(atom->getArgument(pos))) {
721  auto loc = valueIndex.getAggregatorLocation(*agg);
722  op->addCondition(std::unique_ptr<RamCondition>(new RamBinaryRelation(
723  BinaryConstraintOp::EQ, std::unique_ptr<RamValue>(new RamElementAccess(
724  level, pos, getRelation(atom)->getArg(pos))),
725  std::unique_ptr<RamValue>(
726  new RamElementAccess(loc.level, loc.component, loc.name)))));
727  }
728  }
729 
730  // TODO: support constants in nested records!
731  } else if (const AstRecordInit* rec = dynamic_cast<const AstRecordInit*>(cur)) {
732  // add an unpack level
733  const Location& loc = valueIndex.getDefinitionPoint(*rec);
734  op = std::unique_ptr<RamOperation>(
735  new RamLookup(std::move(op), loc.level, loc.component, rec->getArguments().size()));
736 
737  // add constant constraints
738  for (size_t pos = 0; pos < rec->getArguments().size(); ++pos) {
739  if (AstConstant* c = dynamic_cast<AstConstant*>(rec->getArguments()[pos])) {
740  op->addCondition(std::unique_ptr<RamCondition>(new RamBinaryRelation(
741  BinaryConstraintOp::EQ, std::make_unique<RamElementAccess>(level, pos),
742  std::make_unique<RamNumber>(c->getIndex()))));
743  }
744  }
745  } else {
746  std::cout << "Unsupported AST node type: " << typeid(*cur).name() << "\n";
747  assert(false && "Unsupported AST node for creation of scan-level!");
748  }
749  }
750

752  for (const auto& cur : valueIndex.getVariableReferences()) {
753  // the first appearance
754  const Location& first = *cur.second.begin();
755  // all other appearances
756  for (const Location& loc : cur.second) {
757  if (first != loc) {
758  op->addCondition(std::unique_ptr<RamCondition>(new RamBinaryRelation(BinaryConstraintOp::EQ,
759  std::unique_ptr<RamValue>(
760  new RamElementAccess(first.level, first.component, first.name)),
761  std::unique_ptr<RamValue>(
762  new RamElementAccess(loc.level, loc.component, loc.name)))));
763  }
764  }
765  }
766 
767  /* add conditions caused by atoms, negations, and binary relations */
768  for (const auto& lit : clause.getBodyLiterals()) {
769  // for atoms
770  if (dynamic_cast<const AstAtom*>(lit)) {
771  // covered already within the scan/lookup generation step
772 
773  // for binary relations
774  } else if (auto binRel = dynamic_cast<const AstConstraint*>(lit)) {
775  std::unique_ptr<RamValue> valLHS = translateValue(binRel->getLHS(), valueIndex);
776  std::unique_ptr<RamValue> valRHS = translateValue(binRel->getRHS(), valueIndex);
777  op->addCondition(std::unique_ptr<RamCondition>(
778  new RamBinaryRelation(binRel->getOperator(), translateValue(binRel->getLHS(), valueIndex),
779  translateValue(binRel->getRHS(), valueIndex))));
780 
781  // for negations
782  } else if (auto neg = dynamic_cast<const AstNegation*>(lit)) {
783  // get contained atom
784  const AstAtom* atom = neg->getAtom();
785 
786  // create constraint
787  RamNotExists* notExists = new RamNotExists(getRelation(atom));
788 
789  auto arity = atom->getArity();
790 
791  // account for two extra provenance columns
792  if (Global::config().has("provenance")) {
793  arity -= 2;
794  }
795 
796  for (size_t i = 0; i < arity; i++) {
797  const auto& arg = atom->getArgument(i);
798  // for (const auto& arg : atom->getArguments()) {
799  notExists->addArg(translateValue(*arg, valueIndex));
800  }

803  if (Global::config().has("provenance")) {
804  notExists->addArg(nullptr);
805  notExists->addArg(nullptr);
806  }
807

809  op->addCondition(std::unique_ptr<RamCondition>(notExists));
810  } else {
811  std::cout << "Unsupported node type: " << typeid(*lit).name();
812  assert(false && "Unsupported node type!");
813  }
814  }
815

817  return std::make_unique<RamInsert>(clause, std::move(op));

941 /** generate RAM code for recursive relations in a strongly-connected component */
942 std::unique_ptr<RamStatement> AstTranslator::translateRecursiveRelation(
943  const std::set<const AstRelation*>& scc, const AstProgram* program,
944  const RecursiveClauses* recursiveClauses, const TypeEnvironment& typeEnv) {
945  // initialize sections
946  std::unique_ptr<RamStatement> preamble;
947  std::unique_ptr<RamSequence> updateTable(new RamSequence());
948  std::unique_ptr<RamStatement> postamble;
949 
950  // --- create preamble ---
951 
952  // mappings for temporary relations
953  std::map<const AstRelation*, std::unique_ptr<RamRelation>> rrel;
954  std::map<const AstRelation*, std::unique_ptr<RamRelation>> relDelta;
955  std::map<const AstRelation*, std::unique_ptr<RamRelation>> relNew;
956 
957  /* Compute non-recursive clauses for relations in scc and push
958  the results in their delta tables. */
959  for (const AstRelation* rel : scc) {
960  std::unique_ptr<RamStatement> updateRelTable;
961 
962  /* create two temporary tables for relaxed semi-naive evaluation */
963  auto relName = getRelationName(rel->getName());
964  rrel[rel] = getRamRelation(rel, &typeEnv, relName, rel->getArity());
965  relDelta[rel] = getRamRelation(rel, &typeEnv, "delta_" + relName, rel->getArity(), true);
966  relNew[rel] = getRamRelation(rel, &typeEnv, "new_" + relName, rel->getArity(), true);
967 
968  /* create update statements for fixpoint (even iteration) */
969  appendStmt(updateRelTable,
970  std::unique_ptr<RamStatement>(new RamSequence(
971  std::unique_ptr<RamStatement>(
972  new RamMerge(std::unique_ptr<RamRelation>(rrel[rel]->clone()),
973  std::unique_ptr<RamRelation>(relNew[rel]->clone()))),
974  std::unique_ptr<RamStatement>(
975  new RamSwap(std::unique_ptr<RamRelation>(relDelta[rel]->clone()),
976  std::unique_ptr<RamRelation>(relNew[rel]->clone()))),
977  std::make_unique<RamClear>(std::unique_ptr<RamRelation>(relNew[rel]->clone())))));
978 
979  /* measure update time for each relation */
980  if (Global::config().has("profile")) {
981  std::ostringstream ost, osn;
982  ost << "@c-recursive-relation;" << rel->getName() << ";" << rel->getSrcLoc() << ";";
983  updateRelTable = std::make_unique<RamLogTimer>(std::move(updateRelTable), ost.str());
984  }
985 
986  /* drop temporary tables after recursion */
987  appendStmt(postamble, std::unique_ptr<RamStatement>(new RamSequence(
988  std::unique_ptr<RamStatement>(new RamDrop(
989  std::unique_ptr<RamRelation>(relDelta[rel]->clone()))),
990  std::unique_ptr<RamStatement>(new RamDrop(
991  std::unique_ptr<RamRelation>(relNew[rel]->clone()))))));
992 
993  /* Generate code for non-recursive part of relation */
994  appendStmt(preamble, translateNonRecursiveRelation(*rel, program, recursiveClauses, typeEnv));
995 
996  /* Generate merge operation for temp tables */
997  appendStmt(preamble, std::unique_ptr<RamStatement>(
998  new RamMerge(std::unique_ptr<RamRelation>(relDelta[rel]->clone()),
999  std::unique_ptr<RamRelation>(rrel[rel]->clone()))));
1000 
1001  /* Add update operations of relations to parallel statements */
1002  updateTable->add(std::move(updateRelTable));
1003  }
1004 
1005  // --- build main loop ---
1006 
1007  std::unique_ptr<RamParallel> loopSeq(new RamParallel());
1008 
1009  // create a utility to check SCC membership
1010  auto isInSameSCC = [&](
1011  const AstRelation* rel) { return std::find(scc.begin(), scc.end(), rel) != scc.end(); };
1012 
1013  /* Compute temp for the current tables */
1014  for (const AstRelation* rel : scc) {
1015  std::unique_ptr<RamStatement> loopRelSeq;
1016 
1017  /* Find clauses for relation rel */
1018  for (size_t i = 0; i < rel->clauseSize(); i++) {
1019  AstClause* cl = rel->getClause(i);
1020 
1021  // skip non-recursive clauses
1022  if (!recursiveClauses->recursive(cl)) {
1023  continue;
1024  }
1025 
1026  // each recursive rule results in several operations
1027  int version = 0;
1028  const auto& atoms = cl->getAtoms();
1029  for (size_t j = 0; j < atoms.size(); ++j) {
1030  const AstAtom* atom = atoms[j];
1031  const AstRelation* atomRelation = getAtomRelation(atom, program);
1032 
1033  // only interested in atoms within the same SCC
1034  if (!isInSameSCC(atomRelation)) {
1035  continue;
1036  }
1037 
1038  // modify the processed rule to use relDelta and write to relNew
1039  std::unique_ptr<AstClause> r1(cl->clone());
1040  r1->getHead()->setName(relNew[rel]->getName());
1041  r1->getAtoms()[j]->setName(relDelta[atomRelation]->getName());
1042  r1->addToBody(std::unique_ptr<AstLiteral>(
1043  new AstNegation(std::unique_ptr<AstAtom>(cl->getHead()->clone()))));
1044 
1045  // replace wildcards with variables (reduces indices when wildcards are used in recursive
1046  // atoms)
1047  nameUnnamedVariables(r1.get());
1048 
1049  // reduce R to P ...
1050  for (size_t k = j + 1; k < atoms.size(); k++) {
1051  if (isInSameSCC(getAtomRelation(atoms[k], program))) {
1052  AstAtom* cur = r1->getAtoms()[k]->clone();
1053  cur->setName(relDelta[getAtomRelation(atoms[k], program)]->getName());
1054  r1->addToBody(std::make_unique<AstNegation>(std::unique_ptr<AstAtom>(cur)));
1055  }
1056  }
1057 
1058  std::unique_ptr<RamStatement> rule = translateClause(*r1, program, &typeEnv, version);
1059 
1060  /* add logging */
1061  if (Global::config().has("profile")) {
1062  std::string clauseText = stringify(toString(*cl));
1063  std::ostringstream line;
1064  line << "recursive-rule;" << rel->getName() << ";";
1065  line << version << ";";
1066  line << cl->getSrcLoc() << ";";
1067  line << clauseText << ";";
1068  std::string label = line.str();
1069  rule = std::unique_ptr<RamStatement>(new RamSequence(
1070  std::unique_ptr<RamStatement>(new RamLogTimer(std::move(rule), "@t-" + label)),
1071  std::unique_ptr<RamLogSize>(new RamLogSize(
1072  std::unique_ptr<RamRelation>(relNew[rel]->clone()), "@n-" + label))));
1073  }
1074 
1075  // add debug info
1076  std::ostringstream ds;
1077  ds << toString(*cl) << "\nin file ";
1078  ds << cl->getSrcLoc();
1079  rule = std::make_unique<RamDebugInfo>(std::move(rule), ds.str());
1080 
1081  // add to loop body
1082  appendStmt(loopRelSeq, std::move(rule));
1083 
1084  // increment version counter
1085  version++;
1086  }
1087  assert(cl->getExecutionPlan() == nullptr || version > cl->getExecutionPlan()->getMaxVersion());
1088  }
1089 
1090  // if there was no rule, continue
1091  if (!loopRelSeq) {
1092  continue;
1093  }
1094 
1095  // label all versions
1096  if (Global::config().has("profile")) {
1097  std::ostringstream line;
1098  line << "recursive-relation;" << rel->getName() << ";" << rel->getSrcLoc() << ";";
1099  std::string label = line.str();
1100  loopRelSeq = std::make_unique<RamLogTimer>(std::move(loopRelSeq), "@t-" + label);
1101  appendStmt(
1102  loopRelSeq, std::make_unique<RamLogSize>(
1103  std::unique_ptr<RamRelation>(relNew[rel]->clone()), "@n-" + label));
1104  }
1105 
1106  /* add rule computations of a relation to parallel statement */
1107  loopSeq->add(std::move(loopRelSeq));
1108  }
1109 
1110  /* construct exit conditions for odd and even iteration */
1111  auto addCondition = [](std::unique_ptr<RamCondition>& cond, std::unique_ptr<RamCondition> clause) {
1112  cond = ((cond) ? std::unique_ptr<RamCondition>(new RamAnd(std::move(cond), std::move(clause)))
1113  : std::move(clause));
1114  };
1115 
1116  std::unique_ptr<RamCondition> exitCond;
1117  for (const AstRelation* rel : scc) {
1118  addCondition(exitCond, std::unique_ptr<RamCondition>(
1119  new RamEmpty(std::unique_ptr<RamRelation>(relNew[rel]->clone()))));
1120  }
1121 
1122  /* construct fixpoint loop */
1123  std::unique_ptr<RamStatement> res;
1124  if (preamble) appendStmt(res, std::move(preamble));
1125  if (!loopSeq->getStatements().empty() && exitCond && updateTable) {
1126  appendStmt(res, std::unique_ptr<RamStatement>(new RamLoop(std::move(loopSeq),
1127  std::make_unique<RamExit>(std::move(exitCond)), std::move(updateTable))));
1128  }
1129  if (postamble) {
1130  appendStmt(res, std::move(postamble));
1131  }
1132  if (res) return res;
1133 
1134  assert(false && "Not Implemented");
1135  return nullptr;
1136 }

1240 std::unique_ptr<RamProgram> AstTranslator::translateProgram(const AstTranslationUnit& translationUnit) {

1242  translationUnit.getAnalysis<TypeEnvironmentAnalysis>()->getTypeEnvironment();
1243 
1244  RecursiveClauses* recursiveClauses = translationUnit.getAnalysis<RecursiveClauses>();
1245 
1246  /* start with an empty sequence */
1247  std::unique_ptr<RamStatement> res = std::make_unique<RamSequence>();
1248 
1249  /* Compute SCCs of program */
1250  const auto& sccGraph = *translationUnit.getAnalysis<SCCGraph>();
1251  const auto& schedule = translationUnit.getAnalysis<RelationSchedule>()->schedule();
1252 
1253  /** Do nothing for empty schedules */
1254  if (schedule.empty()) {
1255  return std::make_unique<RamProgram>(std::move(res));
1256  }
1257 
1258  /* proceed over each step, in stratification these become subprograms */
1259  unsigned index = 0;
1260  for (const RelationScheduleStep& step : schedule) {
1261  std::unique_ptr<RamStatement> current;
1262 
1263  /* during stratification, create and load all predecessor relations in another scc */
1264 
1265  for (const AstRelation* rel : step.computed()) {
1266  createAndLoad(current, rel, typeEnv, true, sccGraph.isRecursive(rel), true);
1267  }
1268 
1269  /* translate the body, this is where actual computation happens */
1270  std::unique_ptr<RamStatement> stmt;
1271  if (!step.recursive()) {
1272  ASSERT(step.computed().size() == 1 && "SCC contains more than one relation");
1273  const AstRelation* rel = *step.computed().begin();
1274  /* Run non-recursive evaluation */
1275  stmt = translateNonRecursiveRelation(
1276  *rel, translationUnit.getProgram(), recursiveClauses, typeEnv);
1277  } else {
1278  stmt = translateRecursiveRelation(
1279  step.computed(), translationUnit.getProgram(), recursiveClauses, typeEnv);
1280  }
1281  appendStmt(current, std::move(stmt));
1282 
1283  /* store all relations with fault tolerance, and output relations without */
1284  for (const AstRelation* rel : step.computed()) {
1285  printSizeStore(current, rel, typeEnv, true);
1286  }
1287 
1288  /* drop expired relations, or all relations for stratification */
1289  if (!Global::config().has("provenance")) {
1290  for (const AstRelation* rel : step.expired()) {
1291  appendStmt(current, std::make_unique<RamDrop>(getRamRelation(rel, &typeEnv)));
1292  }
1293  if (index == schedule.size() - 1) {
1294  for (const AstRelation* rel : step.computed()) {
1295  appendStmt(current, std::make_unique<RamDrop>(getRamRelation(rel, &typeEnv)));
1296  }
1297  }
1298  }
1299 
1300  // append the current step to the result
1301  if (current) {
1302  appendStmt(res, std::move(current));
1303  // increment the index
1304  index++;
1305  }
1306  }
1307 
1308  if (res && Global::config().has("profile")) {
1309  res = std::make_unique<RamLogTimer>(std::move(res), "@runtime;");
1310  }
1311 
1312  // done for main prog
1313  std::unique_ptr<RamProgram> prog(new RamProgram(std::move(res)));
1314 
1315  // add subroutines for each clause
1316  if (Global::config().has("provenance")) {
1317  visitDepthFirst(translationUnit.getProgram()->getRelations(), [&](const AstClause& clause) {
1318  std::stringstream relName;
1319  relName << clause.getHead()->getName();
1320 
1321  if (relName.str().find("@info") != std::string::npos || clause.getBodyLiterals().size() == 0) {
1322  return;
1323  }
1324

1326  relName.str() + "_" + std::to_string(clause.getClauseNum()) + "_subproof";
1327  prog->addSubroutine(
1328  subroutineLabel, makeSubproofSubroutine(clause, translationUnit.getProgram(), typeEnv));
1329  });
1330  }
