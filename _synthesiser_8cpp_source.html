78  /**
79  * Instance method for getIdentifier above.
80  */
81  const std::string identifier(const std::string& name) {
82  auto it = identifiers.find(name);
83  if (it != identifiers.end()) {
84  return it->second;
85  }
86  // strip leading numbers
87  unsigned int i;
88  for (i = 0; i < name.length(); ++i) {
89  if (isalnum(name.at(i)) || name.at(i) == '_') {
90  break;
91  }
92  }
93  std::string id;
94  for (auto ch : std::to_string(identifiers.size() + 1) + '_' + name.substr(i)) {
95  // alphanumeric characters are allowed
96  if (isalnum(ch)) {
97  id += ch;
98  }
99  // all other characters are replaced by an underscore, except when
100  // the previous character was an underscore as double underscores
101  // in identifiers are reserved by the standard
102  else if (id.size() == 0 || id.back() != '_') {
103  id += '_';
104  }
105  }
106  // most compilers have a limit of 2048 characters (if they have a limit at all) for
107  // identifiers; we use half of that for safety
108  id = id.substr(0, 1024);
109  identifiers.insert(std::make_pair(name, id));
110  return id;
111  }
112 
113  // The map of identifiers.
114  std::map<const std::string, const std::string> identifiers;
115 };

1256 
1257  // collect all used indices
1258  IndexMap indices;
1259  visitDepthFirst(prog, [&](const RamNode& node) {
1260  if (const RamScan* scan = dynamic_cast<const RamScan*>(&node)) {

1265  indices[agg->getRelation()].setHashmap(agg->getRelation().isHashmap());

1267  }

1269  indices[ne->getRelation()].setHashmap(ne->getRelation().isHashmap());
1270  indices[ne->getRelation()].addSearch(ne->getKey());
1271  }
1272  });
1273 
1274  // compute smallest number of indices (and report)

1282  *report << "\tNumber of Scan Patterns: " << cur.second.getSearches().size() << "\n";
1283  for (auto& cols : cur.second.getSearches()) {

1285  for (uint32_t i = 0; i < cur.first.getArity(); i++) {
