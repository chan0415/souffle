45  assert(x >= 0 && x < width && "wrong x dimension");
46  assert(y >= 0 && y < height && "wrong y dimension");
47  assert(x + s.length() <= width && "string too long");

122  // place node and its sub-trees
123  void place(uint32_t x, uint32_t y) {
124  // there must exist at least one kid
125  assert(children.size() > 0 && "no children");
126 
127  // set x/y pos
128  xpos = x;
129  ypos = y;
130 
131  height = 0;
132 
133  // compute size of bounding box
134  for (const std::unique_ptr<TreeNode>& k : children) {
135  k->place(x, y + 2);
136  x += k->getWidth() + 1;
137  width += k->getWidth() + 1;
138  height = std::max(height, k->getHeight());
139  }
140  height += 2;
141 
142  // text of inner node is longer than all its sub-trees
143  if (width < txt.length()) {
144  width = txt.length();
145  }
146  };
147

149  void render(ScreenBuffer& s) {
150  s.write(xpos + (width - txt.length()) / 2, ypos, txt);
151  for (const std::unique_ptr<TreeNode>& k : children) {
152  k->render(s);
153  }
